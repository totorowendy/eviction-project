```python
def get_x_y(df, features=['poverty_rate'], label='eviction_rate'):
    #features = ['pct_af_am']
    X = df[features]
    y = df[label]
    #
    #print(X.shape, len(y))
    return X, y
```


```python
def make_plot(df, features=['poverty_rate'], label='eviction_rate'):
    X, y = get_x_y(df, features, label)
    plt.plot(X, y, '.')
```
